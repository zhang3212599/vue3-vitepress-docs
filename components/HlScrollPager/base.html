  <div style="height: 320px;">
    <div ref="xTable" scroll="true" style="max-height:20px; overflow-y:auto; overflow-x:hidden;"><p style="height:100px;">这是一个表格列表</p></div>
    <p>分页：</p>
    <hl-scroll-pager
      ref="scrollPagerRef"
      :current-page="infoData.page"
      :page-size="infoData.page_size"
      :total="total"
      :update-list="onCallBack"
    >
      <el-button type="primary" size="small">插槽内容</el-button>
    </hl-scroll-pager>
  </div>

import { ref } from 'vue'
// 获取ref
const xTable: any = ref<HTMLElement | null>(null)
const scrollPagerRef: any = ref<HTMLElement | null>(null)

const infoData = {
  page: 1,
  page_size: 15
}
const total = 200

const onCallBack = (obj) => {
  console.log('回调出来的值')
  console.log(obj)
}
</script>
