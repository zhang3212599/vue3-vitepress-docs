hl-polymer-input 下拉选择带输入输入框组件 #

基础用法 #

<template>
  <hl-polymer-input
    ref="polymerInputRef"
    :model="infoData"
    :options="optionsList"
  />
</template>
<script setup lang="ts">
import { ref } from 'vue'
const polymerInputRef: any = ref<HTMLElement | null>(null)
const infoData = {}
const optionsList = [
  { label: "全部", value: 0 },
  { label: "选择第一项", value: 1 },
  { label: "选择第二项", value: 2 },
  { label: "选择第三项", value: 3 }
]
</script>
